ก่อนสคริปต์นี้ const stories = []; // <-- ตอนนี้เป็นค่าว่าง (ไม่มี JSON) // Elements const storyGrid = document.getElementById("storyGrid"); const noResult = document.getElementById("noResult"); const selectionModal = document.getElementById("selectionModal"); const modalTitle = document.getElementById("modalTitle"); const modalImage = document.getElementById("modalImage"); const btnSynopsis = document.getElementById("btnSynopsis"); const btnFullStory = document.getElementById("btnFullStory"); // Utils function safeText(v) { return (v ?? "").toString(); } function normalizeText(s) { return safeText(s).trim().toLowerCase(); } function truncate(text, maxLen = 18) { const t = safeText(text); return t.length > maxLen ? t.slice(0, maxLen - 1) + "…" : t; } function getFallbackCover(title) { // สร้างรูป placeholder แบบ gradient (ไม่พึ่งรูปภายนอก) const svg = encodeURIComponent(` `); return `data:image/svg+xml;charset=utf-8,${svg}`; } // Render cards function renderStories(keyword = "") { const q = normalizeText(keyword); const filtered = stories.filter(s => normalizeText(s.title).includes(q)); storyGrid.innerHTML = ""; if (!filtered.length) { noResult.classList.toggle("hidden", stories.length !== 0); // ถ้ามี data แล้วค้นไม่เจอค่อยโชว์ if (stories.length === 0) { // ถ้าไม่มี data เลย ให้โชว์ข้อความแบบสุภาพ noResult.classList.remove("hidden"); noResult.innerHTML = `

ยังไม่มีข้อมูลใน stories (คุณตัด JSON ออกแล้ว) — วางข้อมูลแล้วรีเฟรชได้เลย

`; } return; } noResult.classList.add("hidden"); filtered.forEach(story => { const cover = story.image ? story.image : getFallbackCover(story.title); const card = document.createElement("button"); card.type = "button"; card.className = "w-full max-w-[140px] sm:max-w-[160px] text-left group"; card.onclick = () => openModal(story); card.innerHTML = `
${truncate(story.title, 22)}
${safeText(story.category || "ชาดก")}
`; storyGrid.appendChild(card); }); } // Modal handlers function openModal(story) { const cover = story.image ? story.image : getFallbackCover(story.title); modalTitle.textContent = safeText(story.title || "ชื่อเรื่อง"); modalImage.src = cover; // ปุ่มลิงก์ (ถ้าไม่มีให้ปิดปุ่ม/เปลี่ยนเป็น #) const syn = safeText(story.synopsisUrl || "#"); const full = safeText(story.fullStoryUrl || "#"); btnSynopsis.href = syn; btnFullStory.href = full; // ถ้าเป็น # ให้ disable สไตล์นิดนึง btnSynopsis.classList.toggle("opacity-50", syn === "#"); btnSynopsis.classList.toggle("pointer-events-none", syn === "#"); btnFullStory.classList.toggle("opacity-50", full === "#"); btnFullStory.classList.toggle("pointer-events-none", full === "#"); selectionModal.classList.remove("hidden"); selectionModal.classList.add("flex"); document.body.style.overflow = "hidden"; } function closeModal() { selectionModal.classList.add("hidden"); selectionModal.classList.remove("flex"); document.body.style.overflow = ""; } function closeModalOnOutside(event) { // ปิดเมื่อคลิกพื้นหลัง (นอกกล่อง modal) if (event.target === selectionModal) closeModal(); } // Initial render renderStories("");
